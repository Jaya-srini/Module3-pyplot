import matplotlib.pyplot as plt
from random import randint

# sequence of 10 random integers
y = [randint(1, 100) for _ in range(10)]
x = list(range(len(y)))

# Create a figure with two subplots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 5))

# Scatter plot
ax1.scatter(x, y, color='red')
ax1.set_title("Scatter Plot of Random Numbers")
ax1.set_xlabel("Index")
ax1.set_ylabel("Random Value")

# Line plot
ax2.plot(x, y, color='lightblue')
ax2.set_title("Line Plot of Random Numbers")
ax2.set_xlabel("Index")
ax2.set_ylabel("Random Value")

# Show the plots
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
from random import randint

# sequence of 10 random integers
y = [randint(1, 100) for _ in range(10)]
x = list(range(len(y)))

# Create a figure with two subplots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 5))

# Scatter plot
ax1.scatter(x, y, color='pink', marker='*', s=100, edgecolor='black', alpha=0.7)
ax1.set_title("Scatter Plot of Random Numbers")
ax1.set_xlabel("Index")
ax1.set_ylabel("Random Value")

# Line plot
ax2.plot(x, y, color='purple', marker='o', markersize=8, linestyle=':', linewidth=2, alpha=0.8)
ax2.set_title("Line Plot of Random Numbers")
ax2.set_xlabel("Index")
ax2.set_ylabel("Random Value")

# Show the plots
plt.tight_layout()
plt.show()
